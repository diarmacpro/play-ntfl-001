function reloadOnce() {
  const reloaded = localStorage.getItem('hasReloaded');

  if (!reloaded) {
    localStorage.setItem('hasReloaded', 'true');
    location.reload();
  }
}

// Panggil fungsi ini saat dibutuhkan
